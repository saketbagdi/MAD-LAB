public class Main
{
	public static void main(String[] args) {
		int n = Integer.parseInt(args[0]);
		for (int i=1;i<=10;i++)
		{
		    System.out.print(n);
		    System.out.print(" x ");
		    System.out.print(i);
		    System.out.print(" = ");
		    System.out.print(n*i);
		    System.out.println();
		}
	}
}
